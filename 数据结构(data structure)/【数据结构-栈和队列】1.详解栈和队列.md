class MyStack {
public:
    /** Initialize your data structure here. */
    MyStack() {

    }
    
    /** Push element x onto stack. */
    void push(int x) {
		queue<int> tempQ;
		tempQ.push(x);
		while(!data.empty()) {
			tempQ.push(data.front());
			data.pop();
		}
		while(!tempQ.empty()){
			data.push(tempQ.front());
			tempQ.pop();
		}
    }
    
    /** Removes the element on top of the stack and returns that element. */
    int pop() {
		int ans = data.front();
		data.pop();
		return ans;
    }
    
    /** Get the top element. */
    int top() {
		return data.front();
    }
    
    /** Returns whether the stack is empty. */
    bool empty() {
		return data.empty();
    }
private:
	queue<int> data; 
};
